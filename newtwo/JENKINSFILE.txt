pipeline{
    agent {
        label 'master'
    }
    stages{
        stage('BUILD')
        {
         steps{
          sh 'virtualenv --no-site-packages .env'
          sh '. .env/bin/activate'
          sh 'pip install -r requirements.txt'

         }
        }
        stage('RUN') {
            steps{
                configFileProvider([configFile(fileId: 'SCRIPT_FILE', variable: 'Envfile')]) {
                    sh '''
                    chmod u+x $Envfile
                    . $Envfile
                    echo \$DB
                    python newtwo/bin/Maincode.py
                    '''
                }
                }
            }
         }
}
