pipeline{
    agent {
        label 'master'
    }
    stages{
        stage('BUILD')
        {
         steps{
          sh 'virtualenv --no-site-packages .env'
          sh '. .env/bin/activate'
          if (fileExists('newtwo/requirements.txt')){
            sh 'pip install -r requirements.txt'
           }
         }
        }
        stage('RUN') {
            steps{
                sh "echo hi"
                sh "python newtwo/bin/Maincode.py"
                }              
            }
         }
}
